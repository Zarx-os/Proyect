void push()

{

    struct node *nptr = malloc(sizeof(struct node));

    nptr -> data = item;

    nptr -> next = top;

    top = nptr;

}


void pop()

{

    if(top == NULL)

         printf("\n\nStack is empty");

    else

    {

         struct mode *temp;

         temp = top;

         top = top -> next;

         printf("\n\n%d deleted", temp -> data);

         free(temp);

    }

}
